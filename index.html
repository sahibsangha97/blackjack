  <script>

    // --- Customize this text if you ever want to change it ---

    const DEALER_STAND_CARD_TEXT = "I hate you, Diya :)";

    // ---------------------------------------------------------


    const dealerHandEl = document.getElementById("dealerHand");

    const playerHandEl = document.getElementById("playerHand");

    const dealerChip = document.getElementById("dealerChip");

    const playerChip = document.getElementById("playerChip");

    const statusText = document.getElementById("statusText");

    const toast = document.getElementById("toast");


    const dealBtn = document.getElementById("dealBtn");

    const hitBtn = document.getElementById("hitBtn");

    const standBtn = document.getElementById("standBtn");

    const resetBtn = document.getElementById("resetBtn");


    let gameStarted = false;

    let hitIndex = 0; // controls "Be" -> "My" -> "Valentine?"


    const hitSequence = ["Be", "My", "Valentine?"];


    function clearHands(){

      dealerHandEl.innerHTML = "";

      playerHandEl.innerHTML = "";

      dealerChip.textContent = "Hand: 0 cards";

      playerChip.textContent = "Hand: 0 cards";

    }


    function updateChips(){

      dealerChip.textContent = `Hand: ${dealerHandEl.children.length} cards`;

      playerChip.textContent = `Hand: ${playerHandEl.children.length} cards`;

    }


    function setToast(main, sub){

      toast.innerHTML = "";

      if(main) toast.appendChild(document.createTextNode(main));

      if(sub){

        const s = document.createElement("small");

        s.textContent = sub;

        toast.appendChild(s);

      }

    }


    function createCard({cornerTop="?", center="?", suit="", isRed=false, isBack=false}){

      const card = document.createElement("div");

      card.className = "card" + (isBack ? " back" : "");

      if(isBack){

        const pattern = document.createElement("div");

        pattern.className = "pattern";

        card.appendChild(pattern);

      }


      const tl = document.createElement("div");

      tl.className = "corner" + (isRed ? " red" : "");

      tl.textContent = cornerTop;


      const br = document.createElement("div");

      br.className = "corner br" + (isRed ? " red" : "");

      br.textContent = cornerTop;


      const centerText = document.createElement("div");

      centerText.className = "centerText" + (isRed ? " red" : "");

      centerText.textContent = center;


      card.appendChild(tl);

      card.appendChild(br);

      card.appendChild(centerText);


      if(suit){

        const suitEl = document.createElement("div");

        suitEl.className = "suit" + (isRed ? " red" : "");

        suitEl.textContent = suit;

        card.appendChild(suitEl);

      }


      return card;

    }


    function deal(){

      if(gameStarted) return;

      gameStarted = true;

      hitIndex = 0;


      clearHands();


      // Player: "Will", "You"

      playerHandEl.appendChild(createCard({cornerTop:"W", center:"Will"}));

      playerHandEl.appendChild(createCard({cornerTop:"Y", center:"You"}));


      // Dealer: Q♥

      dealerHandEl.appendChild(createCard({cornerTop:"Q", center:"Q", suit:"♥", isRed:true}));


      updateChips();


      statusText.textContent = "Player’s turn";

      setToast("You have 2 cards.", "Hit for more cards… or Stand if you’re scared.");


      dealBtn.disabled = true;

      hitBtn.disabled = false;

      standBtn.disabled = false;

    }


    function hit(){

      if(!gameStarted) return;


      if(hitIndex >= hitSequence.length){

        hitBtn.disabled = true;

        setToast("No more hits available.", "The deck has… feelings.");

        return;

      }


      const word = hitSequence[hitIndex];

      playerHandEl.appendChild(createCard({cornerTop: word[0].toUpperCase(), center: word}));

      hitIndex++;


      updateChips();


      if(hitIndex === hitSequence.length){

        hitBtn.disabled = true;

        statusText.textContent = "Soft-lock achieved ✅";

        setToast("Okay that’s enough cards.", "Your hand is now a complete sentence.");

      } else {

        setToast("Nice.", "Keep hitting… if you dare.");

      }

    }


    function stand(){

      if(!gameStarted) return;


      // Dealer draws the "message" card

      dealerHandEl.appendChild(createCard({cornerTop:"!", center: DEALER_STAND_CARD_TEXT, isBack:false}));

      updateChips();


      statusText.textContent = "Dealer’s turn (rigged)";

      setToast("Dealer stands.", "Not sure what the rules are anymore.");

      // After stand, lock controls

      hitBtn.disabled = true;

      standBtn.disabled = true;

    }


    function reset(){

      gameStarted = false;

      hitIndex = 0;

      clearHands();

      updateChips();

      statusText.textContent = "Waiting for player…";

      setToast("", "");

      dealBtn.disabled = false;

      hitBtn.disabled = true;

      standBtn.disabled = true;

    }


    dealBtn.addEventListener("click", deal);

    hitBtn.addEventListener("click", hit);

    standBtn.addEventListener("click", stand);

    resetBtn.addEventListener("click", reset);
